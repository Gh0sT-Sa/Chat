        extractUserData: function(el) {
            const name = el.getAttribute("n")?.trim() || el.querySelector('.u-topic')?.textContent || "مجهول";
            const icon = el.querySelector('.u-ico')?.src || "";
            const hash = el.querySelector('.uhash')?.textContent || "مخفي";
            
            // 🛠️ حل مشكلة 64x64: سحب الصورة الأصلية بإزالة لواحق التصغير
            let pic = el.getAttribute('data-pic'); 
            if (!pic) {
                const picDiv = el.querySelector('.u-pic, .tiger-pic, .v-pic');
                if (picDiv) {
                    const bg = picDiv.style.backgroundImage || window.getComputedStyle(picDiv).backgroundImage;
                    const match = bg.match(/url\(["']?(.*?)["']?\)/);
                    if (match && match[1]) pic = match[1];
                }
            }
            if(!pic) {
                pic = "pic/unknown.png";
            } else {
                // إزالة الإضافات التي تضعها السيرفرات لتصغير الصورة
                pic = pic.replace(/_mini/g, '').replace(/_th/g, '').replace(/_64x64/g, '').replace(/_50x50/g, '');
                if (!pic.startsWith('http')) pic = window.location.origin + '/' + pic.replace(/^\/+/, '');
            }
            
            return { name, pic, icon, hash };
        },

                this.repairCore(".v-user, .uzr", "#ff5722");
                
                if (this.settings.cameraSpy) {
                    document.querySelectorAll('.v-cam-active, i.fa-video-camera').forEach(cam => {
                        const userEl = cam.closest('.uzr, .v-user');
                        if (userEl && !userEl._nexusCamAlert) {
                            const name = userEl.getAttribute('n');
                            this.toast(`🔔 رادار الكاميرا: <b style="color:#00ffcc">${name}</b> يقوم بالبث/المشاهدة الآن.`, 4000);
                            userEl._nexusCamAlert = true;
                        }
                    });
                }
            } else { // JawalHost Universal
                document.querySelectorAll('.btn, .minix, [id^="ahmed"], [id^="sultan"]').forEach(el => el.style.setProperty('display', 'inline-block', 'important'));
                document.querySelectorAll('[class^="itarr"], .nonot, .ubnr').forEach(el => el.remove());
                this.repairCore(".uzr", "#4caf50");
            }

            if (this.settings.profileSniper) {
                const myName = window.myN || "";
                document.querySelectorAll('.log-msg').forEach(msg => {
                    if (msg.textContent.includes('فتح ملف') && msg.textContent.includes(myName) && !msg._nexusSniper) {
                        this.toast(`👁️ كاشف المراقبة: شخص ما يتصفح بروفايلك الآن!`, 5000);
                        msg._nexusSniper = true;
                    }
                });
            }
        },

        repairCore: function(targetSelector, themeColor) {
            document.querySelectorAll(targetSelector).forEach(el => {
                const rawId = el.getAttribute('id') || el.getAttribute('n') || "";
                const cleanId = rawId.replace(/\D/g, '');

                const isGhost = el.querySelector('img.ustat[src*="s4.png"]');
                const style = window.getComputedStyle(el);
                
                // 🛠️ الإظهار النووي لقهر إخفاء جوال هوست المعقد (إجبار CSS)
                if (style.fontSize === '0px' || style.display === 'none' || style.visibility === 'hidden' || isGhost) {
                    el.style.cssText += 'display: block !important; visibility: visible !important; opacity: 1 !important; height: auto !important; width: 100% !important; position: relative !important; z-index: 99 !important;';
                    
                    const topic = el.querySelector('.u-topic');
                    if (topic) {
                        topic.style.cssText += 'display: inline-block !important; font-size: 14px !important; visibility: visible !important;';
                    }
                    
                    if (!el.getAttribute('data-nexus-spy')) {
                        el.style.setProperty('border-right', '3px solid #ff3e3e', 'important');
                        el.style.setProperty('background-color', 'rgba(255, 62, 62, 0.1)', 'important');
                    }
                    el.setAttribute('data-nexus-spy', '1');
                }

                if ((el.getAttribute('data-nexus-spy') === '1' || isGhost) && !el._nexusClickBound) {
                    el.style.cursor = "pointer";
                    el.addEventListener("click", (e) => {
                        e.stopPropagation();
                        if (typeof upro === "function") upro(cleanId);
                        if (typeof openw === "function") setTimeout(() => openw(cleanId, true), 100);
                    });
                    el._nexusClickBound = true;
                }

                // تمرير المعرفات للقائمة الذكية
                this.initSmartMenu(el, cleanId, rawId);

                const adminCls = ['__rv_me', 'ahmed', 'sultan', 'tiger-admin', 'admin-hidden', 'custom-alaw', 'noshow'];
                adminCls.forEach(c => { if (el.classList.contains(c)) el.classList.remove(c); });
                
                const n = el.getAttribute('n'), p = el.getAttribute('data-pic');
                const pic = el.querySelector('.u-pic');
                const topicFallback = el.querySelector('.u-topic');
                if (topicFallback && n && topicFallback.textContent.trim() === "") topicFallback.textContent = n;
                if (pic && p && !pic.style.backgroundImage) pic.style.backgroundImage = `url("${p}")`;
            });
            this.checkHiddenRadar();
        },

        startGlobalObserver: function() {
            let timeout = null;
            new MutationObserver(() => {
                const searchBox = document.getElementById('usearch');
                if (searchBox && searchBox.value.trim().length > 0) return;

                clearTimeout(timeout);
                timeout = setTimeout(() => {
                    this.runAdapter();
                    this.updateFakeCounter();
                }, 150); // تم تخفيف الضغط ليكون 150 ملي ثانية لتخفيف العبء على المتصفح

            }).observe(document.body, { childList: true, subtree: true });
        }
    };

    // 🛠️ تصدير كائن السكربت للمتغير العام لكي تستطيع أزرار القائمة الذكية الوصول إليه
    window.Nexus_Ghost_Ref = Nexus_Ghost;
    Nexus_Ghost.init();
})();
