console.log("Gh0sT-KSA V4.9 | القائمة الذكية العائمة لجميع الأعضاء");

(function() {
    // 1) فحص الجهاز وإعداد الذاكرة
    const isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
    if (window.ghostKsaActive) window.ghostObserver && window.ghostObserver.disconnect();
    window.ghostKsaActive = true;
    window.alertedGhosts = window.alertedGhosts || new Set();

    // 2) مسح الألوان الجانبية (بدون تمييز أحمر)
    const fixStyle = document.createElement('style');
    fixStyle.innerHTML = `
        .uzr { border-right: none !important; border-left: none !important; cursor: pointer !important; } 
        /* تصميم القائمة الذكية */
        .ghost-smart-menu {
            position: absolute; background: #2c3e50; color: #fff; padding: 5px; border-radius: 8px;
            box-shadow: 0 4px 15px rgba(0,0,0,0.5); z-index: 999999; display: flex; flex-direction: column; gap: 5px;
            min-width: 130px; font-family: Tahoma, sans-serif; font-size: 13px; direction: rtl;
        }
        .ghost-menu-item {
            padding: 8px 12px; cursor: pointer; border-radius: 5px; transition: background 0.2s; text-align: center;
        }
        .ghost-menu-item:hover { background: #34495e; }
        .ghost-menu-close {
            background: #c0392b; color: white; border-radius: 5px; text-align: center; padding: 5px; cursor: pointer; font-weight: bold; margin-top: 3px;
        }
    `;
    document.head.appendChild(fixStyle);

    // إغلاق أي قائمة مفتوحة عند النقر في مكان آخر
    document.addEventListener('click', (e) => {
        if (!e.target.closest('.ghost-smart-menu') && !e.target.closest('.uzr')) {
            document.querySelectorAll('.ghost-smart-menu').forEach(m => m.remove());
        }
    });

    // 3) إظهار الأعضاء
    function showUsers() {
        document.querySelectorAll('.btn').forEach(el => el.style.display = 'block');
        document.querySelectorAll('[class^="itarr"]').forEach(el => el.remove());
        if (typeof _ma56zznz2 !== 'undefined' && Array.isArray(_ma56zznz2)) {
            _ma56zznz2.forEach(obj => {
                document.querySelectorAll(`.${obj.cls}`).forEach(el => {
                    el.classList.remove(obj.cls); el.style.display = 'block';
                });
            });
        }
    }
    
    // 4) حاوية التنبيهات
    if (!document.getElementById("mobile-toast-container")) {
        const container = document.createElement("div");
        container.id = "mobile-toast-container";
        container.style = "position:fixed;top:15px;left:50%;transform:translateX(-50%);z-index:99999;max-width:320px;text-align:center";
        document.body.appendChild(container);
    }
    
    function showAlert(user) {
        const toast = document.createElement("div");
        toast.onclick = () => toast.remove(); 
        toast.style = "background:#f0f0f0;border:2px solid #333;border-radius:10px;padding:10px;margin-top:10px;max-width:280px;font-family:sans-serif;color:#000;box-shadow:0 2px 6px rgba(0,0,0,.2);cursor:pointer;text-align:left;animation: slideDown 0.3s ease;";
        toast.innerHTML = `
          <div style="font-weight:bold;text-align:center;margin-bottom:8px">🔔 تنبيه رادار</div>
          <div style="display:flex;align-items:center;gap:8px">
            <img src="${user.pic}" style="width:32px;height:32px;border-radius:50%;border:1px solid #ccc;object-fit:cover;">
            <div style="flex-grow:1">
              <div style="font-weight:bold">${user.name}</div>
              <div style="font-size:12px;color:gray">${user.hash}</div>
            </div>
          </div>
          <div style="margin-top:10px;padding:8px;background:#e0e0e0;border-radius:6px;text-align:center;font-weight:bold;color:#c0392b;">دخل مخفي للروم</div>
        `;
        document.getElementById("mobile-toast-container").appendChild(toast);
        setTimeout(() => toast.remove(), 6000); 
    }
    
    // 5) كشف الرتب العليا
    function showAllHigherRanksOnlyHidden() {
        document.querySelectorAll('.uzr').forEach(el => {
            const isHiddenStyle = el.style.maxHeight === '0px' || el.style.display === 'none';
            const hasS4 = el.querySelector('img.ustat[src*="s4.png"]');
            const hasHiddenClass = el.classList.contains('__rv_me') || el.classList.contains('me');

            if (isHiddenStyle || hasS4 || hasHiddenClass) {
                el.style.setProperty('display', 'block', 'important');
                el.style.setProperty('max-height', 'none', 'important');
                el.style.setProperty('height', 'auto', 'important');
                el.style.setProperty('opacity', '1', 'important');
                el.classList.remove('__rv_me', 'me');
                
                el.querySelectorAll('[style*="display: none"]').forEach(inner => {
                    inner.style.setProperty('display', 'block', 'important');
                });
                
                const ustat = el.querySelector("img.ustat, img[src*='s0.png']");
                if (ustat && !ustat.src.includes('s4.png') && (hasHiddenClass || isHiddenStyle)) {
                    ustat.src = 'imgs/s4.png';
                }
            }
        });
    }
    
    // 6) إصلاح الإطارات
    function fixHiddenFrames() {
        document.querySelectorAll('.uzr').forEach(el => {
            const style = window.getComputedStyle(el);
            if (style.width === '0px' || style.height === '0px') {
                el.style.width = ''; el.style.height = '';
            }
            const frameImg = el.querySelector('.u-pic img[class^="itarr_"]');
            if (frameImg) {
                frameImg.classList.forEach(cls => {
                    if (cls.startsWith('itarr_')) {
                        const frameName = cls.replace('itarr_', '').toLowerCase();
                        if (el.classList.contains(frameName)) el.classList.remove(frameName);
                    }
                });
            }
            ['ahmed', 'mhmood', '__rv_me'].forEach(cls => {
                if (el.classList.contains(cls)) { el.classList.remove(cls); el.style.width = ''; el.style.height = ''; }
            });
            const topic = el.querySelector('.u-topic');
            if (topic) {
                if (window.getComputedStyle(topic).backgroundColor === window.getComputedStyle(topic).color) topic.style.color = '#000';
            }
        });
        document.querySelectorAll(".uzr.custom-alaw").forEach(el => el.classList.remove("custom-alaw"));
    }
    
    // 7) فحص التنبيهات
    function checkHiddenUsers() {
        document.querySelectorAll('.uzr.inroom').forEach(userEl => {
            const statusImg = userEl.querySelector('img.ustat');
            if (statusImg && statusImg.src.includes('s4.png')) {
                const uidClass = [...userEl.classList].find(c => c.startsWith('uid'));
                const userId = uidClass ? uidClass.slice(3) : null;
                
                if (userId && !window.alertedGhosts.has(userId)) {
                    window.alertedGhosts.add(userId); 
                    const name = userEl.getAttribute("n")?.trim() || "بدون اسم";
                    let pic = "pic/unknown.png";
                    const picDiv = userEl.querySelector('.u-pic');
                    if (picDiv) {
                        const match = (picDiv.style.backgroundImage || window.getComputedStyle(picDiv).backgroundImage).match(/url\(["']?(.*?)["']?\)/);
                        if (match && match[1]) pic = match[1].startsWith('http') ? match[1] : window.location.origin + '/' + match[1].replace(/^\/+/, '');
                    }
                    showAlert({ name, pic, hash: userEl.querySelector('.uhash')?.textContent || "" });
                }
            }
        });
    }

    // 8) نظام القائمة الذكية لجميع الأعضاء
    function enableSmartMenu() {
        document.querySelectorAll('.uzr').forEach(user => {
            user.style.setProperty('pointer-events', 'auto', 'important');

            if (!user._menuBound) {
                user.addEventListener("click", function(e) {
                    // منع فتح البروفايل أو الخاص الافتراضي
                    e.stopImmediatePropagation();
                    e.preventDefault();
                    
                    // مسح أي قائمة مفتوحة مسبقاً
                    document.querySelectorAll('.ghost-smart-menu').forEach(m => m.remove());

                    // استخراج البيانات
                    const uidClass = [...user.classList].find(c => c.startsWith("uid"));
                    const userId = uidClass ? uidClass.slice(3) : null;
                    const userName = user.getAttribute("n")?.trim() || "بدون اسم";
                    let encryptedId = "غير متوفر";

                    // محاولة استخراج المعرف المشفر (upro)
                    const picEl = user.querySelector('.u-pic');
                    if (picEl) {
                        const onclickStr = picEl.getAttribute('onclick');
                        if (onclickStr) {
                            const match = onclickStr.match(/upro\(['"]([^'"]+)['"]\)/);
                            if (match && match[1]) encryptedId = match[1];
                        }
                    }

                    // بناء القائمة
                    const menu = document.createElement('div');
                    menu.className = 'ghost-smart-menu';
                    
                    // تحديد موقع القائمة بناءً على مكان النقرة
                    menu.style.left = `${e.pageX + 10}px`;
                    menu.style.top = `${e.pageY}px`;

                    menu.innerHTML = `
                        <div class="ghost-menu-item" id="btn-profile">👤 فتح البروفايل</div>
                        <div class="ghost-menu-item" id="btn-private">💬 فتح الخاص</div>
                        <div class="ghost-menu-item" id="btn-copy-id">🔑 نسخ المعرف</div>
                        <div class="ghost-menu-item" id="btn-copy-name">📝 نسخ الاسم</div>
                        <div class="ghost-menu-close">إغلاق ❌</div>
                    `;

                    document.body.appendChild(menu);

                    // وظائف الأزرار
                    document.getElementById('btn-profile').onclick = () => {
                        if (picEl && picEl.getAttribute('onclick')) {
                            try { eval(picEl.getAttribute('onclick')); } catch(err) {}
                        }
                        menu.remove();
                    };
                    
                    document.getElementById('btn-private').onclick = () => {
                        if (userId && typeof window.openw === 'function') window.openw(userId, true);
                        menu.remove();
                    };
                    
                    document.getElementById('btn-copy-id').onclick = () => {
                        navigator.clipboard.writeText(encryptedId);
                        alert(`تم نسخ المعرف المشفر:\n${encryptedId}`);
                        menu.remove();
                    };
                    
                    document.getElementById('btn-copy-name').onclick = () => {
                        navigator.clipboard.writeText(userName);
                        alert(`تم نسخ الاسم:\n${userName}`);
                        menu.remove();
                    };
                    
                    menu.querySelector('.ghost-menu-close').onclick = () => menu.remove();

                }, true);
                
                user._menuBound = true;
            }
        });
    }
    
    // 9) المراقبة
    window.ghostObserver = new MutationObserver(() => {
        const searchBox = document.getElementById('usearch');
        if (searchBox && searchBox.value.trim().length > 0) return;
        
        showUsers();
        showAllHigherRanksOnlyHidden();
        fixHiddenFrames();
        checkHiddenUsers();
        enableSmartMenu();
    });
    
    window.ghostObserver.observe(document.body, { childList: true, subtree: true });
    
    showUsers();
    showAllHigherRanksOnlyHidden();
    fixHiddenFrames();
    checkHiddenUsers();
    enableSmartMenu();
    
    const startupToast = document.createElement("div");
    startupToast.style = "background:#2980b9; color:white; padding:12px 25px; border-radius:8px; position:fixed; bottom:20px; right:20px; z-index:100000; font-family:sans-serif; font-weight:bold; box-shadow:0 4px 15px rgba(0,0,0,0.4); border-right: 4px solid #fff; direction: rtl;";
    startupToast.innerHTML = `🛡️ <b>Gh0sT-KSA V4.9 Activated</b><br><span style="font-size:12px;">اضغط على أي عضو لفتح القائمة الذكية</span>`;
    document.body.appendChild(startupToast);
    setTimeout(() => startupToast.remove(), 4000);

})();
