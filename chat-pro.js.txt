(function() {
    'use strict';

    const Nexus_Ghost = {
        config: {
            name: "Nexus Ghost Interface",
            author: "Gh0sT-Sa",
            version: "1.5.0"
        },
        env: { 
            device: "Detecting...", 
            host: "Unknown", 
            isJawalHost: false,
            isAuthorized: true 
        },
        checkList: { domReady: false, engineFound: false, adapterLoaded: false },

        init: function() {
            this.detectEnv();
            this.identifyEngine();
            this.renderConsoleArt();
            this.runAdapter();
            this.startGlobalObserver();
        },

        detectEnv: function() {
            const ua = navigator.userAgent;
            if (/android/i.test(ua)) this.env.device = "Android";
            else if (/iPhone|iPad|iPod/i.test(ua)) this.env.device = "iOS";
            else if (/Windows/i.test(ua)) this.env.device = "Windows";
            else this.env.device = "Linux/Other";
            this.checkList.domReady = true;
        },

        identifyEngine: function() {
            const h = window.location.hostname.toLowerCase();
            
            // نظام التعرف الذكي على "نواة جوال هوست" عبر البصمات البرمجية
            const jawalHostFingerprints = [
                window.jawalHost,
                typeof _ma56zznz2 !== 'undefined',
                typeof _m56zznz2 !== 'undefined',
                document.querySelector('script[src*="x3.js"]'),
                h.includes('ksa') || h.includes('njd') || h.includes('alkhalij') || h.includes('7alij')
            ];

            if (jawalHostFingerprints.some(f => f)) {
                this.env.host = "JawalHost_Universal";
                this.env.isJawalHost = true;
                this.checkList.engineFound = true;
            } else if (window.phoneHost) {
                this.env.host = "PhoneHost";
            } else if (document.querySelector('script[src*="tiger"]')) {
                this.env.host = "TigerHost";
            }
        },

        adapters: {
            JawalHost_Universal: {
                run: function() { this.applyUniversalBypass(); },
                refresh: function() { this.applyUniversalBypass(); },
                applyUniversalBypass: function() {
                    // 1) إظهار الأزرار المخفية (تجميع كافة معرفات الخليج والمملكة ونجد واليمن)
                    // تشمل: المعرفات التي تبدأ بـ sultan, ahmed, khaleej وكلاسات minix و btn
                    const universalSelectors = [
                        '.btn', '.minix', '.mini', 
                        '[id^="ahmed"]', '[id^="sultan"]', '[id^="khaleej"]', 
                        '[style*="display: none"]', '[style*="visibility: hidden"]'
                    ].join(',');

                    document.querySelectorAll(universalSelectors).forEach(el => {
                        // استثناء العناصر التي لا نريد العبث بها (مثل عناصر النظام الأساسية)
                        if (el.classList.contains('u-pic') || el.id === 'l1' || el.id === 'l2') return;

                        el.style.setProperty('display', 'block', 'important');
                        el.style.setProperty('visibility', 'visible', 'important');
                        el.style.setProperty('opacity', '1', 'important');
                        el.style.setProperty('max-height', 'none', 'important');
                        el.style.setProperty('z-index', '999999', 'important');
                    });

                    // 2) إزالة الإطارات المقيدة (itarr) نهائياً
                    document.querySelectorAll('[class^="itarr"]').forEach(el => el.remove());

                    // 3) معالجة المصفوفات الغامضة (Ma و M) المكتشفة في كافة النسخ
                    ['_ma56zznz2', '_m56zznz2'].forEach(matrixName => {
                        if (typeof window[matrixName] !== 'undefined' && Array.isArray(window[matrixName])) {
                            window[matrixName].forEach(obj => {
                                if (obj.cls) {
                                    document.querySelectorAll(`.${obj.cls}`).forEach(el => {
                                        el.classList.remove(obj.cls);
                                        el.style.setProperty('display', 'block', 'important');
                                    });
                                }
                            });
                        }
                    });

                    // 4) كشف المستخدمين المخفيين (Bypass Hidden Users)
                    const userSelector = ".uzr[style*='max-height: 0'], .uzr[style*='max-height:0'], .uzr.hidden, .uzr[style*='display:none']";
                    document.querySelectorAll(userSelector).forEach(el => {
                        el.style.setProperty('display', 'block', 'important');
                        el.style.setProperty('max-height', 'none', 'important');
                        el.style.setProperty('visibility', 'visible', 'important');

                        // منع سكريبت الشات من إعادة الوسم عبر Patching
                        if (!el._nexus_patched) {
                            const originalAdd = el.classList.add;
                            el.classList.add = function(...args) {
                                const forbidden = ['__rv_me', 'hidden', 'ahmed', 'mhmood', 'sultan'];
                                const filtered = args.filter(cls => !forbidden.includes(cls));
                                return originalAdd.apply(this, filtered);
                            };
                            el._nexus_patched = true;
                        }

                        // تصحيح أيقونة المتواجدين (ustat)
                        const ustat = el.querySelector("img.ustat");
                        if (ustat && !ustat.src.includes('s4.png')) {
                            ustat.src = 'imgs/s4.png';
                        }
                    });

                    // 5) إصلاح شفافية وتطابق ألوان النصوص (Topic & Hash Correction)
                    document.querySelectorAll('.u-topic, .uhash').forEach(item => {
                        const style = window.getComputedStyle(item);
                        if (style.color === style.backgroundColor || style.opacity === '0' || style.display === 'none') {
                            item.style.setProperty('color', '#000', 'important');
                            item.style.setProperty('opacity', '1', 'important');
                            item.style.setProperty('display', 'inline-block', 'important');
                        }
                    });
                }
            }
        },

        runAdapter: function() {
            const activeHost = this.env.host;
            if (this.adapters[activeHost]) {
                this.adapters[activeHost].run();
                this.checkList.adapterLoaded = true;
            }
        },

        renderConsoleArt: function() {
            const style = "color: #00ffff; background: #111; padding: 5px; border-radius: 5px; font-family: monospace;";
            console.log(`%c >>> ${this.config.name} v${this.config.version} BY ${this.config.author} <<< `, style);
            console.log(`%c Target Engine: JawalHost Universal Kernel `, "color: #ffaa00; font-weight: bold;");
            console.table(this.checkList);
        },

        startGlobalObserver: function() {
            const observer = new MutationObserver(() => {
                if (this.env.isJawalHost && this.adapters.JawalHost_Universal) {
                    this.adapters.JawalHost_Universal.refresh();
                }
            });
            observer.observe(document.body, { childList: true, subtree: true });
        }
    };

    Nexus_Ghost.init();
})();
