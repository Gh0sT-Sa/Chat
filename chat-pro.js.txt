console.log("Gh0sT-KSA Chameleon V3.8 | نظام الألوان والرصد الموحد");

(function() {
    const alertedUsers = new Set();

    // 1) حقن CSS الألوان الذكي (نظام الألوان الذي طلبته)
    const styleEl = document.createElement('style');
    styleEl.innerHTML = `
        /* الحالة الافتراضية للرتب المخفية s4 (الأزرق) */
        .uzr[data-status="hidden"] { 
            display: block !important; max-height: none !important; visibility: visible !important; opacity: 1 !important;
            border-right: 4px solid #3498db !important; background: rgba(52, 152, 219, 0.05);
        }
        /* الحالة: مغلق الخاص (الأحمر) */
        .uzr[data-status="private-closed"] { border-right: 4px solid #e74c3c !important; }
        
        /* تنسيق عام لقائمة المتواجدين لسهولة التمييز */
        .uzr[data-ghost="true"] * { pointer-events: none !important; }
        
        /* حاوية التنبيهات الموحدة */
        #ghost-alert-container { position:fixed; top:20px; left:50%; transform:translateX(-50%); z-index:1000000; width: 320px; }
        .ghost-toast { 
            background:#fff; border-right:6px solid #3498db; padding:15px; margin-bottom:10px; border-radius:12px; 
            box-shadow:0 8px 25px rgba(0,0,0,0.3); direction:rtl; font-family:sans-serif; cursor:pointer;
            animation: slideIn 0.3s ease; transition: 0.3s;
        }
        @keyframes slideIn { from { transform: translateY(-20px); opacity: 0; } to { transform: translateY(0); opacity: 1; } }
    `;
    document.head.appendChild(styleEl);

    // 2) محرك فحص الحالة والألوان
    function applyColorLogic(el) {
        const ustat = el.querySelector('img.ustat');
        if (!ustat) return;

        const src = ustat.src;
        const isS4 = src.includes('s4.png'); // مخفي
        const isClosedPrivate = el.querySelector('.u-topic.p-closed, .u-topic[style*="color: red"]') || el.classList.contains('private-off'); 

        if (isS4) {
            el.setAttribute('data-status', 'hidden');
            el.style.color = "#3498db"; // اسم المخفي باللون الأزرق
        } else if (isClosedPrivate) {
            el.setAttribute('data-status', 'private-closed');
            el.style.color = "#e74c3c"; // اسم مغلق الخاص بالأحمر
        }
        
        // الألوان الأساسية الأخرى
        if (src.includes('s1.png')) el.style.borderRight = "4px solid #2ecc71"; // أخضر (نشط)
        if (src.includes('s2.png')) el.style.borderRight = "4px solid #f1c40f"; // أصفر (خامل)
        if (src.includes('s0.png')) el.style.borderRight = "4px solid #95a5a6"; // رمادي (خرج/معلق)
    }

    // 3) نظام التنبيه الموحد (بضغطة واحدة للإغلاق)
    function showSingleAlert(user) {
        let container = document.getElementById("ghost-alert-container");
        if (!container) {
            container = document.createElement("div");
            container.id = "ghost-alert-container";
            document.body.appendChild(container);
        }

        const t = document.createElement("div");
        t.className = "ghost-toast";
        t.onclick = () => t.remove(); // إغلاق بضغطة واحدة
        
        const locText = user.isInRoom ? "دخل لغرفتك الآن 🚪⚡" : "متواجد في الدردشة (مخفي) 🔍";
        const themeColor = user.isInRoom ? "#e74c3c" : "#3498db";
        t.style.borderRightColor = themeColor;

        t.innerHTML = `
            <div style="display:flex; align-items:center; gap:12px;">
                <img src="${user.pic}" style="width:45px; height:45px; border-radius:50%; border:2px solid ${themeColor}; object-fit:cover;">
                <div style="flex-grow:1">
                    <div style="font-weight:bold; color:#2c3e50; font-size:14px;">${user.name}</div>
                    <div style="font-size:11px; color:#7f8c8d;">رقم العضوية: <span style="color:${themeColor}">${user.hash}</span></div>
                    <div style="font-size:12px; color:${themeColor}; font-weight:bold;">${locText}</div>
                </div>
            </div>
        `;
        container.appendChild(t);
        setTimeout(() => { if(t) t.remove(); }, 7000);
    }

    // 4) اعتراض نظام الموقع (فتح البروفايل)
    const originalOpenw = window.openw;
    window.openw = function(id) {
        const userEl = document.querySelector(`.uid${id}`);
        if (userEl && userEl.querySelector('img.ustat[src*="s4.png"]')) {
            if (typeof uprof === 'function') return uprof(id);
            if (window.uprof) return window.uprof(id);
        }
        if (originalOpenw) return originalOpenw.apply(this, arguments);
    };

    // 5) المحرك الشامل لفك حماية الرتب العليا
    function chameleonScan() {
        document.querySelectorAll(".uzr").forEach(el => {
            const ustat = el.querySelector('img.ustat');
            const isS4 = ustat && ustat.src.includes('s4.png');
            
            if (isS4) {
                // إظهار الرتب العليا مهما كانت وسيلة الإخفاء
                el.style.setProperty('display', 'block', 'important');
                el.style.setProperty('max-height', 'none', 'important');
                el.classList.remove('__rv_me', 'me', 'hidden');
                el.setAttribute('data-ghost', 'true');
                
                // تطبيق نظام الألوان
                applyColorLogic(el);

                // التنبيه (منع التكرار)
                const uid = [...el.classList].find(c => c.startsWith('uid'));
                if (uid && !alertedUsers.has(uid)) {
                    alertedUsers.add(uid);
                    
                    const picDiv = el.querySelector('.u-pic');
                    let pic = "pic/unknown.png";
                    if (picDiv) {
                        let bg = picDiv.style.backgroundImage || window.getComputedStyle(picDiv).backgroundImage;
                        let match = bg.match(/url\(["']?(.*?)["']?\)/);
                        if (match && match[1]) pic = match[1];
                    }

                    showSingleAlert({
                        name: el.getAttribute("n") || "مخفي",
                        hash: el.querySelector('.uhash')?.textContent || "---",
                        pic: pic,
                        isInRoom: el.classList.contains('inroom')
                    });
                }
            } else {
                // تطبيق الألوان للأعضاء العاديين أيضاً
                applyColorLogic(el);
            }
        });
    }

    setInterval(chameleonScan, 1000);

    // رسالة الترحيب
    const welcome = document.createElement("div");
    welcome.style = "position:fixed; bottom:25px; left:25px; background:#2c3e50; color:white; padding:15px; border-radius:10px; z-index:999999; direction:rtl; font-weight:bold; box-shadow:0 10px 30px rgba(0,0,0,0.5); border-right: 5px solid #3498db;";
    welcome.innerHTML = "🦎 <b>Gh0sT-KSA V3.8 [Chameleon]</b><br><span style='font-size:12px; font-weight:normal;'>✔️ نظام الألوان الموحد يعمل<br>✔️ إغلاق التنبيه بضغطة واحدة</span>";
    document.body.appendChild(welcome);
    setTimeout(() => welcome.remove(), 5000);
})();
