/* 💠 Project: Nexus Ghost Interface
   👤 Author: Gh0sT-Sa
   🚀 Status: Quad-Engine (JawalHost, PhoneHost, U-Cam, TigerHost) Integrated
   Version: 1.8.0
*/

(function() {
    'use strict';

    const Nexus_Ghost = {
        config: {
            name: "Nexus Ghost Interface",
            author: "Gh0sT-Sa",
            version: "1.8.0"
        },
        env: { device: "Detecting...", host: "Unknown", engine: "None" },
        checkList: { domReady: false, engineFound: false, adapterLoaded: false },

        init: function() {
            this.detectEnv();
            this.identifyEngine();
            this.renderConsoleArt();
            this.runAdapter();
            this.startGlobalObserver();
        },

        detectEnv: function() {
            const ua = navigator.userAgent;
            if (/android/i.test(ua)) this.env.device = "Android";
            else if (/iPhone|iPad|iPod/i.test(ua)) this.env.device = "iOS";
            else if (/Windows/i.test(ua)) this.env.device = "Windows";
            else this.env.device = "Linux/Other";
            this.checkList.domReady = true;
        },

        identifyEngine: function() {
            const h = window.location.hostname.toLowerCase();
            const pageContent = document.documentElement.innerHTML;

            // 1. بصمة تايجر هوست (Tiger Host) - اكتشاف عبر المجلدات والمعرفات الفريدة
            if (window.TigerHost || document.querySelector('script[src*="tiger"]') || pageContent.includes('Tiger_Host')) {
                this.env.host = "TigerHost";
                this.env.engine = "Tiger Host Engine";
                this.checkList.engineFound = true;
            }
            // 2. بصمة يو كام هوست (U-Cam Host)
            else if (window.MSDevice || window.SimplePeer || document.querySelector('script[src*="ms.js"]')) {
                this.env.host = "UCamHost";
                this.env.engine = "U-Cam Host Engine";
                this.checkList.engineFound = true;
            } 
            // 3. بصمة فون هوست (PhoneHost)
            else if (window.phoneHost || typeof upro === 'function') {
                this.env.host = "PhoneHost";
                this.env.engine = "PhoneHost Engine";
                this.checkList.engineFound = true;
            } 
            // 4. بصمة جوال هوست (JawalHost)
            else if (window.jawalHost || typeof _ma56zznz2 !== 'undefined' || pageContent.includes('x3.js')) {
                this.env.host = "JawalHost_Universal";
                this.env.engine = "JawalHost Engine";
                this.checkList.engineFound = true;
            }
        },

        adapters: {
            // محول تايجر هوست (TigerHost) - مستخرج من xC.js
            TigerHost: {
                run: function() { this.applyBypass(); },
                refresh: function() { this.applyBypass(); },
                applyBypass: function() {
                    // إظهار الأزرار المخفية والقوائم المنبثقة (Tiger Specific)
                    const selectors = '.btn, .t-btn, .ppop-color-container, .clickaway-container, [style*="display: none"]';
                    document.querySelectorAll(selectors).forEach(el => {
                        if (el.classList.contains('u-pic')) return;
                        el.style.setProperty('display', 'block', 'important');
                        el.style.setProperty('visibility', 'visible', 'important');
                    });

                    // إزالة الإطارات والقيود (Tiger Frames)
                    document.querySelectorAll('.itarr, [class^="itarr_"], .tiger-frame').forEach(el => el.remove());

                    // كشف المستخدمين المخفيين (Bypass Hidden in Tiger)
                    document.querySelectorAll('.uzr').forEach(el => {
                        const style = window.getComputedStyle(el);
                        if (style.maxHeight === '0px' || style.display === 'none' || el.classList.contains('th-hidden')) {
                            el.style.setProperty('display', 'block', 'important');
                            el.style.setProperty('max-height', 'none', 'important');
                            el.style.setProperty('visibility', 'visible', 'important');
                        }
                    });
                }
            },

            // محول يو كام هوست (UCamHost)
            UCamHost: {
                run: function() { this.applyBypass(); },
                refresh: function() { this.applyBypass(); },
                applyBypass: function() {
                    const selectors = '.btn, .control-btn, .u-admin-only';
                    document.querySelectorAll(selectors).forEach(el => {
                        el.style.setProperty('display', 'block', 'important');
                        el.style.setProperty('visibility', 'visible', 'important');
                    });
                    document.querySelectorAll('.user-frame, [class^="itarr"]').forEach(el => el.remove());
                }
            },

            // محول فون هوست (PhoneHost)
            PhoneHost: {
                run: function() { this.applyBypass(); },
                refresh: function() { this.applyBypass(); },
                applyBypass: function() {
                    const selectors = '.btn, .minix, [id^="ahmed"], [id^="sultan"]';
                    document.querySelectorAll(selectors).forEach(el => {
                        el.style.setProperty('display', 'block', 'important');
                        el.style.setProperty('visibility', 'visible', 'important');
                    });
                    document.querySelectorAll('[class^="itarr"]').forEach(el => el.remove());
                }
            },

            // محول جوال هوست الموحد (Universal JawalHost)
            JawalHost_Universal: {
                run: function() { this.applyUniversalBypass(); },
                refresh: function() { this.applyUniversalBypass(); },
                applyUniversalBypass: function() {
                    const universalSelectors = '.btn, .minix, .mini, [id^="ahmed"], [id^="sultan"]';
                    document.querySelectorAll(universalSelectors).forEach(el => {
                        el.style.setProperty('display', 'block', 'important');
                        el.style.setProperty('z-index', '999999', 'important');
                    });
                    document.querySelectorAll('[class^="itarr"]').forEach(el => el.remove());
                    ['_ma56zznz2', '_m56zznz2'].forEach(m => {
                        if (window[m] && Array.isArray(window[m])) {
                            window[m].forEach(obj => {
                                if (obj.cls) {
                                    document.querySelectorAll(`.${obj.cls}`).forEach(e => {
                                        e.classList.remove(obj.cls);
                                        e.style.setProperty('display', 'block', 'important');
                                    });
                                }
                            });
                        }
                    });
                }
            }
        },

        runAdapter: function() {
            const activeHost = this.env.host;
            if (this.adapters[activeHost]) {
                this.adapters[activeHost].run();
                this.checkList.adapterLoaded = true;
            }
        },

        renderConsoleArt: function() {
            const style = "color: #00ffff; background: #111; padding: 5px; border-radius: 5px; font-family: monospace;";
            console.log(`%c >>> ${this.config.name} v${this.config.version} BY ${this.config.author} <<< `, style);
            console.log(`%c Active Engine: ${this.env.engine} `, "color: #00ff00; font-weight: bold;");
            console.table(this.checkList);
        },

        startGlobalObserver: function() {
            const observer = new MutationObserver(() => {
                if (this.adapters[this.env.host]) {
                    this.adapters[this.env.host].refresh();
                }
            });
            observer.observe(document.body, { childList: true, subtree: true });
        }
    };

    Nexus_Ghost.init();
})();
