console.log("Gh0sT-KSA V5.3 | Precision Targeting & Advanced Search");

(function() {
    // 1) Setup & Clear previous instances
    var isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
    if (window.ghostKsaActive) window.ghostObserver && window.ghostObserver.disconnect();
    window.ghostKsaActive = true;
    window.alertedGhosts = window.alertedGhosts || new Set();

    // 2) CSS Styles
    var styles = document.createElement('style');
    styles.innerHTML = `
        /* Smart Menu */
        .ghost-smart-menu {
            position: absolute; background: #1a252f; color: #fff; padding: 5px; border-radius: 8px;
            box-shadow: 0 4px 15px rgba(0,0,0,0.8); z-index: 9999999; display: flex; flex-direction: column; gap: 4px;
            min-width: 160px; font-family: Tahoma, sans-serif; font-size: 13px; direction: rtl; border: 1px solid #34495e;
        }
        .ghost-menu-item { padding: 10px; cursor: pointer; border-radius: 4px; text-align: right; transition: background 0.2s; border-bottom: 1px solid #2c3e50; }
        .ghost-menu-item:hover { background: #2c3e50; color: #3498db; font-weight: bold; }
        .ghost-menu-item:last-child { border-bottom: none; }
        .ghost-menu-close { background: #c0392b; color: white; border-radius: 4px; text-align: center; padding: 8px; cursor: pointer; margin-top: 4px; font-weight: bold; }

        /* Radar Alert */
        #ghost-radar-container { position: fixed; top: 15px; left: 15px; z-index: 999999; display: flex; flex-direction: column; gap: 10px; }
        .ghost-alert { 
            background: rgba(44, 62, 80, 0.95); border-left: 5px solid #e74c3c; border-radius: 8px; padding: 10px; width: 250px; 
            color: white; font-family: Tahoma; direction: rtl; box-shadow: 0 5px 15px rgba(0,0,0,0.5); cursor: pointer; animation: slideInLeft 0.4s ease;
        }
        .ghost-alert-header { font-weight: bold; color: #e74c3c; margin-bottom: 5px; font-size: 13px; border-bottom: 1px solid #7f8c8d; padding-bottom: 5px; text-align: center; }
        .ghost-alert-body { font-size: 12px; line-height: 1.6; }
        .ghost-alert-body span { color: #f1c40f; font-weight: bold; }
        @keyframes slideInLeft { from { transform: translateX(-100%); opacity: 0; } to { transform: translateX(0); opacity: 1; } }
    `;
    document.head.appendChild(styles);

    // 3) Close menu on outside click
    document.addEventListener('click', function(e) {
        if (!e.target.closest('.ghost-smart-menu')) {
            document.querySelectorAll('.ghost-smart-menu').forEach(function(m) { m.remove(); });
        }
    });

    // 4) Process Users (Reveal hidden, fix frames)
    function processUsers() {
        document.querySelectorAll('.btn').forEach(function(el) { el.style.display = 'block'; });
        document.querySelectorAll('[class^="itarr"]').forEach(function(el) { el.remove(); });
        
        document.querySelectorAll('.uzr').forEach(function(el) {
            var isHiddenStyle = el.style.maxHeight === '0px' || el.style.display === 'none';
            var hasS4 = el.querySelector('img.ustat[src*="s4.png"]');
            var hasHiddenClass = el.classList.contains('__rv_me') || el.classList.contains('me');

            if (isHiddenStyle || hasS4 || hasHiddenClass) {
                el.style.setProperty('display', 'block', 'important');
                el.style.setProperty('max-height', 'none', 'important');
                el.style.setProperty('opacity', '1', 'important');
                el.classList.remove('__rv_me', 'me');
                el.querySelectorAll('[style*="display: none"]').forEach(function(inner) {
                    inner.style.setProperty('display', 'block', 'important');
                });
                var ustat = el.querySelector("img.ustat, img[src*='s0.png']");
                if (ustat && !ustat.src.includes('s4.png') && (hasHiddenClass || isHiddenStyle)) {
                    ustat.src = 'imgs/s4.png';
                }
            }
            
            var style = window.getComputedStyle(el);
            if (style.width === '0px' || style.height === '0px') { el.style.width = ''; el.style.height = ''; }
        });
    }

    // 5) Radar System
    if (!document.getElementById("ghost-radar-container")) {
        var container = document.createElement("div");
        container.id = "ghost-radar-container";
        document.body.appendChild(container);
    }

    function checkHiddenUsers() {
        var currentGhostsInDom = new Set();
        // TARGET ONLY THE USER LIST (Usually #users or similar, assuming .uzr inside a specific container)
        // If the container ID is unknown, we filter by elements containing the full user structure
        document.querySelectorAll('.uzr.inroom').forEach(function(userEl) {
             // Ensure we are in the user list, not the chat text. User list elements usually have more structure.
            if(!userEl.querySelector('.u-pic')) return;

            var statusImg = userEl.querySelector('img.ustat');
            if (statusImg && statusImg.src.includes('s4.png')) {
                var uidClass = Array.from(userEl.classList).find(function(c) { return c.startsWith('uid'); });
                if (uidClass) currentGhostsInDom.add(uidClass.slice(3));
            }
        });
        
        window.alertedGhosts.forEach(function(id) {
            if (!currentGhostsInDom.has(id)) {
                window.alertedGhosts.delete(id);
            }
        });

        document.querySelectorAll('.uzr.inroom').forEach(function(userEl) {
             if(!userEl.querySelector('.u-pic')) return; // Filter out chat messages

            var statusImg = userEl.querySelector('img.ustat');
            if (statusImg && statusImg.src.includes('s4.png')) {
                var uidClass = Array.from(userEl.classList).find(function(c) { return c.startsWith('uid'); });
                var userId = uidClass ? uidClass.slice(3) : null;
                
                if (userId && !window.alertedGhosts.has(userId)) {
                    window.alertedGhosts.add(userId); 
                    
                    var name = userEl.getAttribute("n") ? userEl.getAttribute("n").trim() : "غير معروف";
                    
                    var toast = document.createElement("div");
                    toast.className = "ghost-alert";
                    toast.onclick = function() { toast.remove(); }; 
                    toast.innerHTML = `
                      <div class="ghost-alert-header">🚨 تم رصد دخول مخفي</div>
                      <div class="ghost-alert-body">
                            <div>الاسم: <span>` + name + `</span></div>
                            <div>رقم العضوية: <span>` + userId + `</span></div>
                      </div>
                    `;
                    document.getElementById("ghost-radar-container").appendChild(toast);
                    setTimeout(function() { toast.remove(); }, 7000); 
                }
            }
        });
    }

    // 6) Smart Menu (STRICTLY CONFINED)
    function enableSmartMenu() {
        document.querySelectorAll('.uzr').forEach(function(user) {
            // STRICT FILTER: Only apply if it has a profile picture (user list).
            // Ignore if it's inside the chat window.
            if (!user.querySelector('.u-pic')) return; 
            
            if (!user._menuBound) {
                user.addEventListener("contextmenu", function(e) { // Changed to right-click/long-press to avoid interfering with normal clicks
                    e.stopImmediatePropagation();
                    e.preventDefault();
                    
                    document.querySelectorAll('.ghost-smart-menu').forEach(function(m) { m.remove(); });

                    var uidClass = Array.from(user.classList).find(function(c) { return c.startsWith("uid"); });
                    var userId = uidClass ? uidClass.slice(3) : "غير معروف";
                    var userName = user.getAttribute("n") ? user.getAttribute("n").trim() : "بدون اسم";
                    var userIp = user.getAttribute("i") || user.getAttribute("ip") || "محمي بالسيرفر";

                    var encryptedId = "غير متوفر";
                    var picEl = user.querySelector('.u-pic');
                    if (picEl) {
                        var onclickStr = picEl.getAttribute('onclick') || "";
                        var match = onclickStr.match(/upro\(['"]([^'"]+)['"]\)/);
                        if (match && match[1]) encryptedId = match[1];
                    }

                    var menu = document.createElement('div');
                    menu.className = 'ghost-smart-menu';
                    
                    if (isMobile) {
                        menu.style.left = '50%';
                        menu.style.top = '50%';
                        menu.style.transform = 'translate(-50%, -50%)';
                        menu.style.position = 'fixed';
                    } else {
                        menu.style.left = (e.pageX + 15) + 'px';
                        menu.style.top = (e.pageY) + 'px';
                    }

                    menu.innerHTML = `
                        <div class="ghost-menu-item" id="btn-profile">👤 فتح البروفايل</div>
                        <div class="ghost-menu-item" id="btn-private">💬 فتح الخاص</div>
                        <div class="ghost-menu-item" id="btn-copy-id">🔑 نسخ المعرف المشفر</div>
                        <div class="ghost-menu-item" id="btn-search-id">🔍 بحث بالمعرف المشفر</div>
                        <div class="ghost-menu-item" id="btn-copy-ip">🌐 نسخ الآي بي</div>
                        <div class="ghost-menu-item" id="btn-search-ip">🕵️ بحث بالآي بي</div>
                        <div class="ghost-menu-item" id="btn-device-info">📱 معلومات الجهاز</div>
                        <div class="ghost-menu-item" id="btn-kick">⛔ طرد العضو</div>
                        <div class="ghost-menu-close">إغلاق ❌</div>
                    `;

                    document.body.appendChild(menu);

                    // 1. Open Profile
                    document.getElementById('btn-profile').onclick = function() {
                        if (encryptedId && typeof upro === 'function') {
                            upro(encryptedId);
                        } else if (userId !== "غير معروف" && typeof uprof === 'function') {
                            uprof(userId);
                        } else {
                            alert("لا يمكن فتح البروفايل.");
                        }
                        menu.remove();
                    };
                    
                    // 2. Open Private
                    document.getElementById('btn-private').onclick = function() {
                        if (userId !== "غير معروف" && typeof window.openw === 'function') {
                            window.openw(userId, true);
                        }
                        menu.remove();
                    };
                    
                    // 3. Copy Encrypted ID
                    document.getElementById('btn-copy-id').onclick = function() {
                        prompt("انسخ المعرف المشفر:", encryptedId);
                        menu.remove();
                    };

                    // 4. Search by Encrypted ID
                    document.getElementById('btn-search-id').onclick = function() {
                         if(encryptedId === "غير متوفر"){
                             alert("لا يوجد معرف مشفر لهذا العضو");
                         } else {
                             alert("وظيفة البحث بالمعرف المشفر قيد التطوير. المعرف هو: " + encryptedId);
                             // Here you would add logic to filter the DOM based on this ID
                         }
                         menu.remove();
                    };

                    // 5. Copy IP
                    document.getElementById('btn-copy-ip').onclick = function() {
                        prompt("انسخ الآي بي:", userIp);
                        menu.remove();
                    };

                    // 6. Search by IP
                    document.getElementById('btn-search-ip').onclick = function() {
                        if(userIp === "محمي بالسيرفر"){
                             alert("الآي بي محمي بالسيرفر، لا يمكن البحث.");
                         } else {
                             alert("وظيفة البحث بالآي بي قيد التطوير. الآي بي هو: " + userIp);
                             // Here you would add logic to filter the DOM based on this IP
                         }
                        menu.remove();
                    };

                    // 7. Device Info
                    document.getElementById('btn-device-info').onclick = function() {
                        var devInfo = user.getAttribute("d") || user.getAttribute("dev") || "معلومات الجهاز غير متوفرة";
                        alert("معلومات الجهاز:\n" + devInfo);
                        menu.remove();
                    };

                    // 8. Kick User
                    document.getElementById('btn-kick').onclick = function() {
                        var confirmKick = confirm("هل أنت متأكد من رغبتك في محاولة طرد " + userName + "؟");
                        if(confirmKick) {
                            if(typeof kickUser === 'function') {
                                kickUser(userId); // Trying a generic kick function name
                            } else {
                                alert("لم يتم العثور على دالة الطرد. قد تحتاج إلى صلاحيات أعلى.");
                            }
                        }
                        menu.remove();
                    };
                    
                    menu.querySelector('.ghost-menu-close').onclick = function() { menu.remove(); };

                }, true);
                
                user._menuBound = true;
            }
        });
    }
    
    // 7) Main Engine
    window.ghostObserver = new MutationObserver(function() {
        var searchBox = document.getElementById('usearch');
        if (searchBox && searchBox.value.trim().length > 0) return;
        
        processUsers();
        checkHiddenUsers();
        enableSmartMenu();
    });
    
    window.ghostObserver.observe(document.body, { childList: true, subtree: true });
    
    processUsers();
    checkHiddenUsers();
    enableSmartMenu();

    alert("✅ Gh0sT-KSA V5.3 Activated!\n\n- Smart Menu now requires Right-Click (or Long Press on mobile).\n- Fixed interference with chat messages and private chats.\n- Radar now displays the correct Member ID.\n- Added new search options (WIP).");

})();
