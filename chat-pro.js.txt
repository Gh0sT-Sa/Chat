                    const bg = picDiv.style.backgroundImage || window.getComputedStyle(picDiv).backgroundImage;

                if ((el.getAttribute('data-nexus-spy') === '1' || isGhost) && !el._nexusClickBound) {
                    el.style.cursor = "pointer";
                    el.addEventListener('click', function(e) {
                        e.stopPropagation();
                        // استدعاء الحدث الافتراضي المبرمج في الشات نفسه إذا وجد jQuery
                        if(typeof $ !== 'undefined' && $(this).data('events')) {
                            $(this).trigger('click');
                        } else {
                            if(typeof upro === 'function') upro(cleanId);
                        }
                    });
                    el._nexusClickBound = true;
                }

                this.initSmartMenu(el, cleanId, rawId);
                
                const n = el.getAttribute('n'), p = el.getAttribute('data-pic');
                const pic = el.querySelector('.u-pic');
                if (pic && p && !pic.style.backgroundImage) pic.style.backgroundImage = `url("${p}")`;
            });
            this.checkHiddenRadar();
        },

        startGlobalObserver: function() {
            let timeout = null;
            new MutationObserver(() => {
                const searchBox = document.getElementById('usearch');
                if (searchBox && searchBox.value.trim().length > 0) return;

                clearTimeout(timeout);
                timeout = setTimeout(() => {
                    this.runAdapter();
                    this.updateFakeCounter();
                }, 250); 
            }).observe(document.body, { childList: true, subtree: true });
        }
    };

    window.Nexus_Ghost_Ref = Nexus_Ghost;
    Nexus_Ghost.init();
})();
